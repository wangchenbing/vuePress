            setInterval(() => {
                this.flag = false
                this.start()
            }, 6000)
        },
        methods: {
            start () {
                this.flag = true
                requestAnimationFrame(() => {
                    this.positionList = this.shuffle(this.positionList)
                })
            },
            shuffle (arr) {
                let i = arr.length;
                while (i) {
                    let j = Math.floor(Math.random() * i--);
                    [arr[j], arr[i]] = [arr[i], arr[j]];
                }
                return arr;
            }
        }
    }
